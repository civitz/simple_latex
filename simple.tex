\newcommand{\singlefigure}[3]{

 \centering

}

\newcommand{\doublefigure}[6]{
\begin{figure}[htbp!]
\centering
\hspace*{-0.2\linewidth}%
\begin{minipage}{.7\textwidth}
  \centering

  \caption{\label{#3}#2}
  
\end{minipage}%

}

}
